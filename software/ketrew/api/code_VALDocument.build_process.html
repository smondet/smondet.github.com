let build_process ?(with_details=false)  =
  let open Log in
  let open Target in
  function
  | `No_operation -> s "No-op"
  | `Long_running (name, content) ->
    s "Long-running " % parens (s name)
    % if with_details
      then s ":" % n %
           indent (concat (
            List.map (Plugin.long_running_log name content)
              ~f:(fun (title, descr) -> s title % s ": " % descr % n)))
      else empty
