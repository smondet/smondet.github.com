Module Ketrew_artifact.Volume (.ml)

module Volume: sig .. end
Module defining “Volumes” which are definitions of a given file structure on a given Ketrew_host.t.
type structure = Ketrew_gen_base_v0.Volume_structure.t
The structure of a volume defines the hierarchy of files relative to a root path, one constructs structures with Ketrew_artifact.Volume.file and Ketrew_artifact.Volume.dir: 
          let src_dir_structure =

            dir "src" [

              dir "lib" [

                file "ketrew.ml";

                file "ketrew_artifact.ml";

                file "ketrew_artifact.mli";

                (* ... *)

              ];

              dir "test" [

                file "main.ml";

              ];

            ]
