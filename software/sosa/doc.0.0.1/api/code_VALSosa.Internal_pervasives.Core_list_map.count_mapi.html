let rec count_mapi ~f l ctr =
      match l with
      | [] -> []
      | [x1] ->
        let f1 = f ctr x1 in
        [f1]
      | [x1; x2] ->
        let f1 = f ctr x1 in
        let f2 = f (ctr + 1) x2 in
        [f1; f2]
      | [x1; x2; x3] ->
        let f1 = f ctr x1 in
        let f2 = f (ctr + 1) x2 in
        let f3 = f (ctr + 2) x3 in
        [f1; f2; f3]
      | [x1; x2; x3; x4] ->
        let f1 = f ctr x1 in
        let f2 = f (ctr + 1) x2 in
        let f3 = f (ctr + 2) x3 in
        let f4 = f (ctr + 3) x4 in
        [f1; f2; f3; f4]
      | x1 :: x2 :: x3 :: x4 :: x5 :: tl ->
        let f1 = f ctr x1 in
        let f2 = f (ctr + 1) x2 in
        let f3 = f (ctr + 2) x3 in
        let f4 = f (ctr + 3) x4 in
        let f5 = f (ctr + 4) x5 in
        f1 :: f2 :: f3 :: f4 :: f5 ::
          (if ctr > 5000
           then mapi_slow ~f ~i:(ctr + 5) tl
           else count_mapi ~f tl (ctr + 5))
